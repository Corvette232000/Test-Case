        // DOM elements
        const stateSelect = document.getElementById('state-select');
        const utilitySelect = document.getElementById('utility-select');
        const calculateBtn = document.getElementById('calculate-btn');
        const resultsDiv = document.getElementById('results');
        const stateIncentivesDiv = document.getElementById('state-incentives');
        const utilityIncentivesDiv = document.getElementById('utility-incentives');
        const stateCountSpan = document.getElementById('state-count');
        const utilityCountSpan = document.getElementById('utility-count');

        // Populate state dropdown
        Object.keys(stateNames).forEach(stateCode => {
            if (incentivesData[stateCode]) {
                const option = document.createElement('option');
                option.value = stateCode;
                option.textContent = stateNames[stateCode];
                stateSelect.appendChild(option);
            }
        });

        // Update utility options based on selected state
        stateSelect.addEventListener('change', function() {
            const selectedState = this.value;
            utilitySelect.innerHTML = '<option value="">-- Select a Utility --</option>';
            
            if (selectedState && incentivesData[selectedState]) {
                // Get unique utilities for the selected state
                const utilities = [...new Set(incentivesData[selectedState]
                    .filter(item => item.Type !== "State")
                    .map(item => item["Utility/Program"]))];
                
                utilities.forEach(utility => {
                    const option = document.createElement('option');
                    option.value = utility;
                    option.textContent = utility;
                    utilitySelect.appendChild(option);
                });
            }
        });

        // Calculate and display incentives
        calculateBtn.addEventListener('click', function() {
            const selectedState = stateSelect.value;
            const selectedUtility = utilitySelect.value;
            
            if (!selectedState || !selectedUtility) {
                alert('Please select both a state and a utility.');
                return;
            }
            
            // Display state incentives
            stateIncentivesDiv.innerHTML = '';
            let stateIncentiveCount = 0;
            
            if (incentivesData[selectedState]) {
                const stateIncentives = incentivesData[selectedState].filter(
                    item => item.Type === "State" && item.Incentive !== "None" && item.Incentive !== "Nope"
                );
                
                stateIncentiveCount = stateIncentives.length;
                
                if (stateIncentiveCount > 0) {
                    stateIncentives.forEach(incentive => {
                        const incentiveCard = document.createElement('div');
                        incentiveCard.className = 'incentive-card state-card';
                        incentiveCard.innerHTML = `
                            <div class="incentive-type">${incentive.Type}</div>
                            <div class="incentive-name">${incentive.Incentive}</div>
                            ${incentive.Link && incentive.Link !== "None" && incentive.Link !== "Nope" ? 
                                `<a href="${incentive.Link}" class="incentive-link" target="_blank">Learn More</a>` : 
                                ''}
                        `;
                        stateIncentivesDiv.appendChild(incentiveCard);
                    });
                } else {
                    stateIncentivesDiv.innerHTML = '<div class="no-incentives">No state incentives available</div>';
                }
            }
            
            // Display utility incentive
            utilityIncentivesDiv.innerHTML = '';
            let utilityIncentiveCount = 0;
            
            if (incentivesData[selectedState]) {
                const utilityIncentives = incentivesData[selectedState].filter(
                    item => item["Utility/Program"] === selectedUtility && 
                           item.Incentive !== "None" && 
                           item.Incentive !== "Nope" &&
                           item.Type !== "State"
                );
                
                utilityIncentiveCount = utilityIncentives.length;
                
                if (utilityIncentiveCount > 0) {
                    utilityIncentives.forEach(incentive => {
                        const incentiveCard = document.createElement('div');
                        incentiveCard.className = 'incentive-card utility-card';
                        incentiveCard.innerHTML = `
                            <div class="incentive-type">${incentive.Type}</div>
                            <div class="incentive-name">${incentive.Incentive}</div>
                            <div class="incentive-description">${incentive["Utility/Program"]}</div>
                            ${incentive.Link && incentive.Link !== "None" && incentive.Link !== "Nope" ? 
                                `<a href="${incentive.Link}" class="incentive-link" target="_blank">Learn More</a>` : 
                                ''}
                        `;
                        utilityIncentivesDiv.appendChild(incentiveCard);
                    });
                } else {
                    utilityIncentivesDiv.innerHTML = '<div class="no-incentives">No utility incentives available</div>';
                }
            }
            
            // Update counts
            stateCountSpan.textContent = stateIncentiveCount;
            utilityCountSpan.textContent = utilityIncentiveCount;
            
            // Show results
            resultsDiv.style.display = 'block';
        });
